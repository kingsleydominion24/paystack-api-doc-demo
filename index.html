4. Frontend Code Example

Here’s how it works in code (sandbox/public key only):

<script src="https://js.paystack.co/v1/inline.js"></script>
<input id="email">
<select id="amount"></select>
<button id="payBtn">Pay</button>

<script>
document.getElementById('payBtn').addEventListener('click', function(){
  let email = document.getElementById('email').value;
  let amount = parseInt(document.getElementById('amount').value);
  let handler = PaystackPop.setup({
    key: 'YOUR_TEST_PUBLIC_KEY',
    email: email,
    amount: amount,
    currency: 'NGN',
    callback: function(response){ alert('Payment complete! Reference: ' + response.reference); },
    onClose: function(){ alert('Payment window closed.'); }
  });
  handler.openIframe();
});
</script>

5. Backend Verification (Recommended)

For real payments, verify transactions using your secret key:

# Node.js Example
const fetch = require('node-fetch');
app.get('/verify/:reference', async (req, res) => {
  const ref = req.params.reference;
  const response = await fetch(`https://api.paystack.co/transaction/verify/${ref}`, {
    headers: { Authorization: 'Bearer YOUR_SECRET_KEY' }
  });
  const data = await response.json();
  res.json(data);
});

6. Tips & Best Practices

  • Always use sandbox/test mode first.
  • Never expose secret keys on the frontend.
  • Verify payments on backend.
  • Keep UI clean and professional.
  • Document your code for learning and sharing.
